  return(x < quantile(x, 0.25) - 1.5 * IQR(x) | x > quantile(x, 0.75) + 1.5 * IQR(x))

#EJERCICO10. Creación de la variable sleep_time a través de una función de cálculo del mismo
total_sleep <- function(x){
  entero <- as.integer(x)
  decimal <- x - entero
  if (decimal >= 0.6){
    #Agregamos +1 hora al entero y evitamos que salgan numeros extremadamente redondos
    #los cuales pueden provocar NA a la hora de convertir
    entero <- entero + 1
    if (entero == 24){
      entero <- 0
      #entero <- entero + 0.01
    }
    #El remanente son los minutos que nos quedan desde que se cumplió la hora
    remanente <- (decimal - 0.6)
    #Se los agregamos al entero
    entero <- entero + remanente
    #Redondeamos a 2 decimales
    return(round(as.double(entero), 3) + 0.001)
  }
  #Si no hay excedente devolvemos el mismo valor redondeado convertido
  #a flotante para evitar NA
  else
    return(round(as.double(x),3) + 0.001)
}

#Creacion de una columna para la conversion de 'dormir'

#Creacion de una columna para la conversion de 'despetar'
for (i in 1:nrow(datos_2)){
  datos_2[i, 'despertar_convert'] <- total_sleep(datos_2[i, 10])

#Vamos a iterar entre las columnas dormir_convert y despertar_convert para obtener
#la diferencia en horas del sueño de cada persona
for (i in 1:nrow(datos_2)){
  #Lo pasamos a un formato legible por la funcion
  dormir <- format(strptime(datos_2[i, 'dormir_convert'], format = '%H.%M', tz = Sys.timezone()), '%H:%M')
  despertar <- format(strptime(datos_2[i, 'despertar_convert'], format = '%H.%M', tz = Sys.timezone()), '%H:%M')
  
  dormir <- as.POSIXct(dormir, format = '%H:%M')
  despertar <- as.POSIXct(despertar, format = '%H:%M')
  #Calculamos la diferencia
  z <- difftime(despertar, dormir, units = 'hours')
  #La agregamos en una nueva columna al dataframe
  datos_2[i, 'sleep_time_h'] <- round(z, 2)
}
